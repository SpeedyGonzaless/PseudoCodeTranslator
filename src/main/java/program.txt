FUNCTION findMaxElem (float[] array) : float BEGIN
	float max <- array[0]
	FOREACH elem IN array
		IF (elem > max) THEN
			max <- elem
		ENDIF
	ENDFOREACH
	RETURN max
END_FUNCTION
